projects/congarevenuecloud/elements/src/lib/tax-breakup/tax-breakup.component.ts

Description

Displays a tax breakdown for a line item.

Usage

import { TaxBreakupModule } from '@congarevenuecloud/elements';

@NgModule({
imports: [TaxBreakupModule, ...]
})
export class AppModule {}
```html
<apt-tax-breakup-popover
[item]="taxBreakup"
[loading]="isTaxLoading"
[error]="taxLoadError"
[productName]="parent?.Product?.Name">
</apt-tax-breakup-popover>

Inputs

error
Type : boolean
Default value : false

When true, displays an error message.
item
Type : TaxBreakup
Default value : null

Tax breakup entry to display.
loading
Type : boolean
Default value : false

When true, displays a loading spinner.
productName
Type : string
Default value : ''

Product name shown as popover subtitle.
title
Type : string
Default value : ''

Optional title text for the popover header. Defaults to the TAX_BREAKUP.TITLE translation key.

Outputs

closePopover
Type : EventEmitter<void>

Emitted when the user clicks the close (×) icon.

 
<div class="px-1 tax-breakup-content">

  <!-- Header -->
  <div class="mb-1 pb-1 border-bottom">
    <div class="d-flex justify-content-between align-items-start">
      <span class="font-weight-bold text-dark">{{ title ? title : ('TAX_BREAKUP.TITLE' | translate) }}</span>
      <button type="button" class="btn btn-link btn-sm p-0 ml-2 text-muted" (click)="closePopover.emit()"
        [attr.aria-label]="'COMMON.CLOSE' | translate">
        <i class="fas fa-times"></i>
      </button>
    </div>
    <small *ngIf="productName" class="text-muted text-break tax-product-name" [attr.title]="productName">{{productName}}</small>
  </div>

    <!-- Loading -->
    <div *ngIf="loading" class="text-center py-2">
      <i class="fas fa-spinner fa-spin text-primary"></i>
    </div>

    <!-- Error -->
    <div *ngIf="error && !loading" class="text-danger py-1">
      <small>{{'TAX_BREAKUP.ERROR' | translate}}</small>
    </div>

    <!-- Empty -->
    <div *ngIf="!loading && !error && !item" class="text-center text-muted py-2">
      <small>{{'TAX_BREAKUP.EMPTY' | translate}}</small>
    </div>

    <!-- Data row -->
    <ng-container *ngIf="!loading && !error && item">
      <div class="tax-entry mb-1 pt-1">
        <div class="d-flex justify-content-between align-items-baseline">
          <span class="text-break mr-2">{{item.TaxType}}</span>
          <span class="text-nowrap">{{(item.TaxAmount ?? 0) | localCurrency | async}}</span>
        </div>
        <div class="text-muted small">
          {{'TAX_BREAKUP.RATE' | translate}}&nbsp;{{item.TaxRate}}%
        </div>
      </div>

      <hr class="my-1">

    <!-- Footer -->
      <div class="d-flex justify-content-between font-weight-bold text-primary">
        <span class="mr-2">{{'TAX_BREAKUP.TOTAL_TAX' | translate}}</span>
        <span class="text-nowrap">{{(item.TaxAmount ?? 0) | localCurrency | async}}</span>
      </div>
    </ng-container>
</div>
